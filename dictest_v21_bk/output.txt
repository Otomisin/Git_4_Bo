import json

def load_dictionary(path):
    """
    Loads the JSON dictionary file and transforms it into a list of dictionaries
    with 'word', 'definition', and 'example' keys.
    """
    try:
        with open(path, 'r', encoding='utf-8') as file:
            raw_data = json.load(file)  # Load JSON as dictionary
            # Transform into a list of dictionaries
            dictionary_data = [
                {"word": key, "definition": value, "example": ""}  # Provide an empty example
                for key, value in raw_data.items()
            ]
            print(f"Loaded {len(dictionary_data)} words.")
            return dictionary_data
    except json.JSONDecodeError as e:
        print(f"JSON decode error: {e}")
    except Exception as e:
        print(f"Error reading the file: {e}")
    return None

        if not dictionary_data:

            return

        for entry in dictionary_data:
            try:
                word_obj, created = Word.objects.get_or_create(

                        'definition': entry['definition'],
                        'example': entry.get('example', "No example available."),  # Default example text

                if created:
                    self.stdout.write(self.style.SUCCESS(f"Added new word: {word_obj.word}"))

            except Exception as e:
                self.stdout.write(self.style.ERROR(f"Error saving word: {entry.get('word', 'Unknown')} - {e}"))

    const searchInput = document.getElementById('search-input');
    const suggestionList = document.getElementById('suggestion-list');
    
    searchInput.addEventListener('input', async () => {
        const query = searchInput.value.trim();
        if (query.length >= 3) {
            const response = await fetch(`/search/suggestions?q=${query}`);
            const suggestions = await response.json();
            displaySuggestions(suggestions);
        } else {
            hideSuggestions();
        }
    });
    
    function displaySuggestions(suggestions) {
        suggestionList.innerHTML = '';
        suggestions.forEach(suggestion => {
            const div = document.createElement('div');
            div.classList.add('list-group-item');
            div.innerHTML = suggestion.html; // Display formatted HTML
            div.addEventListener('click', () => {
                searchInput.value = suggestion.word; // Use plain word for input
                hideSuggestions();
                searchInput.focus();
            });
            suggestionList.appendChild(div);
        });
        suggestionList.style.display = 'block';
    }
    
    function hideSuggestions() {
        suggestionList.style.display = 'none';
    }
    
    // Add click outside listener to hide suggestions
    document.addEventListener('click', (event) => {
        if (!searchInput.contains(event.target) && !suggestionList.contains(event.target)) {
            hideSuggestions();
        }
    });
    
    // Add keyboard navigation
    searchInput.addEventListener('keydown', (event) => {
        const items = suggestionList.getElementsByClassName('list-group-item');
        const activeItem = suggestionList.querySelector('.list-group-item.active');
        
        switch(event.key) {
            case 'ArrowDown':
                event.preventDefault();
                if (!activeItem) {
                    items[0]?.classList.add('active');
                } else {
                    const nextItem = activeItem.nextElementSibling;
                    if (nextItem) {
                        activeItem.classList.remove('active');
                        nextItem.classList.add('active');
                    }
                }
                break;
                
            case 'ArrowUp':
                event.preventDefault();
                if (activeItem) {
                    const prevItem = activeItem.previousElementSibling;
                    if (prevItem) {
                        activeItem.classList.remove('active');
                        prevItem.classList.add('active');
                    }
                }
                break;
                
            case 'Enter':
                if (activeItem) {
                    event.preventDefault();
                    const suggestion = suggestions.find(s => 
                        activeItem.textContent.includes(s.word)
                    );
                    if (suggestion) {
                        searchInput.value = suggestion.word;
                        hideSuggestions();
                    }
                }
                break;
                
            case 'Escape':
                hideSuggestions();
                break;
        }
    });

def home(request):
    query = request.GET.get('q', '').strip()
    words = Word.objects.filter(
        Q(word__icontains=query) | Q(definition__icontains=query)
    ) if query else []

    if query and words.exists():
        # Redirect to the first matching word's detail page
        return redirect('word_detail', pk=words.first().id)

    # Render the home page with the list of matching words
    return render(request, 'dictionary/home.html', {'query': query, 'words': words})

def word_detail(request, pk):
    print(f"Fetching word with ID: {pk}")
    word = get_object_or_404(Word, pk=pk)
    return render(request, 'dictionary/word_detail.html', {'word': word})



def search_results(request):
    # Get the search query from the request
    query = request.GET.get('q', '').strip()

    # Redirect to the home page if the query is empty
    if not query:
        return redirect('home')  # Redirects user back to the homepage

    if query:
        # Fetch the first matching word from the database
        word = Word.objects.filter(word__icontains=query).first()  # Fetch only the first match
        if word:
            # Redirect to the word detail page if a match is found
            return redirect('word_detail', word_id=word.id)
    
    # If no matches are found, render the "No Results" page
    return render(request, 'dictionary/no_results.html', {'query': query})

def random_word(request):
    word = Word.objects.order_by('?').first()  # Fetch a random word
    return render(request, 'dictionary/word_detail.html', {'word': word})
